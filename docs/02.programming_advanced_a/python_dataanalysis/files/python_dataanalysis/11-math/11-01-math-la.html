転置行列とその性質¶

[15]:

X = A
Y = A.T
X @ Y

[15]:

array([[5, 2],
       [2, 2]])

[16]:

Y.T @ X.T

[16]:

array([[5, 2],
       [2, 2]])

[17]:

r2 = np.sqrt(2.0)
V = np.array(
    [
        [-1 / r2, -1 / r2],
        [1 / r2, -1 / r2],
    ]
)
# 1列目と2列目は直交（内積が0）
V[:, 0] @ V[:, 1]

[17]:

0.0

[18]:

# ノルムは1（表示は浮動小数点演算の誤差のため1になっていない）
np.linalg.norm(V[:, 0])

[18]:

0.9999999999999999

[19]:

# 2行2列の単位行列になる
V.T @ V

[19]:

array([[ 1.00000000e+00, -2.23711432e-17],
       [-2.23711432e-17,  1.00000000e+00]])
