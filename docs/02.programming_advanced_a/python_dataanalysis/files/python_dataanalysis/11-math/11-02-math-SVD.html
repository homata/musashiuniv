def sum_up_k_rank(k):
    rec_img = np.zeros((img_n, img_d))
    for i in range(k):
        # i列目だけ
        _U = img_U[:, i].reshape(img_n, -1)
        # i行目だけ
        _Vt = img_Vt[i].reshape(-1, img_d)
        rec_img += img_s[i] * _U @ _Vt
    return rec_img
