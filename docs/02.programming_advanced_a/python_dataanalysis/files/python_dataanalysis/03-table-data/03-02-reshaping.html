groupby_time = tips.groupby("time")[  # ①time列でグルーピング
    # ①グループ化したオブジェクトから列を選択
    ["total_bill", "tip"]
].agg(
    ("mean", "median")
)  # ①平均値と中央値を算出
groupby_time.columns.names = (
    "value",
    "agg",
)  # ②インデックスに名前を設定
