read_csvの便利なオプション¶

csvモジュールを使う¶

[9]:

import csv

with open(
    "data/FEH_00200524_230205124938.csv",
    "r",
    encoding="cp932",
    newline="",
) as f:  # ・・①
    data = csv.reader(f)  # ・・②
    for i, row in enumerate(data):
        for col in row:
            print(col, end="|")
        print()
        print("===")
        if i > 1:
            break
